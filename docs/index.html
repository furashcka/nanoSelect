How use


var select = new NanoSelect(element); //default
var select = new NanoSelect(element, options); //with options

Options


var options = {
    rootClass: 'nanoSelect',
    toggleClass: 'nanoSelect__toggle',
    searchClass: 'nanoSelect__search',
    labelClass: 'nanoSelect__label',
    arrowClass: 'nanoSelect__arrow',
    resultClass: 'nanoSelect__result',
    listClass: 'nanoSelect__list',
    groupClass: 'nanoSelect__group',
    optionClass: 'nanoSelect__option',
    limitClass: 'nanoSelect__limit', //for hidding long text in option (hidding in css)

    openClass: 'nanoSelect_open',
    nativeClass: 'nanoSelect_native',
    positionTopClass: 'nanoSelect_top',
    positionBottomClass: 'nanoSelect_bottom',

    searchPlaceholder: 'Type for search...',
    resultPosition: 'bottom', //top or bottom

    useNative: true, //set - "false" if you want always show only custom select
    search: false
};

Api Methods


var options = {
    opened: function() {}, //called when custom select opened
    closed: function() {}, //called when custom select closed
    changed: function() {}, //called when select changed
    template: function(item) { //template option
        return item.text;
    }
};

select.open(); //open custom select
select.close(); //close custom select

select.set(index, callChange);
/* selecting option from code.
   index (Number)
   callChange (Boolean) - if set false: can't called changed() callback */

select.add(data, pos, callback);
/* add options.
   data (Array) - [
       {
           value: value,
           text: text,
           selected: (Boolean)
       },
       {
           text: groupText,
           children: [
               {
                   value: value,
                   text: text
               }
           ]
       }
   ];
   pos (String) - after or before, default: after
*/

select.remove(indexes); //remove options, '*' - remove all; [0,1,2] - remove by index
select.destroy(); //destroy custom select
